Image Audit Script Template

#!/bin/bash
# Image audit automation script

DOCS_DIR="/path/to/docs"
IMAGES_DIR="$DOCS_DIR/images"
REPORT_DIR="./image_audit_reports"

# Create report directory
mkdir -p "$REPORT_DIR"

# 1. Find all image references in markdown files
echo "üîç Scanning for image references..."
find "$DOCS_DIR" -name "*.md" -exec grep -Ho "!\[[^]]*\]([^)]*)" {} + > "$REPORT_DIR/all_image_refs.txt"

# 2. Extract unique image filenames
echo "üìù Extracting image filenames..."
grep -o "[^/]*\.[a-zA-Z]*)" "$REPORT_DIR/all_image_refs.txt" | sed 's/)$//' | sort -u > "$REPORT_DIR/referenced_images.txt"

# 3. List all existing images
echo "üìÇ Cataloging existing images..."
find "$IMAGES_DIR" -type f -name "*.png" -o -name "*.jpg" -o -name "*.gif" -o -name "*.svg" | xargs basename -a | sort > "$REPORT_DIR/existing_images.txt"

# 4. Find unused images
echo "üóëÔ∏è Identifying unused images..."
comm -23 "$REPORT_DIR/existing_images.txt" "$REPORT_DIR/referenced_images.txt" > "$REPORT_DIR/unused_images.txt"

# 5. Find missing images (404s)
echo "‚ùå Identifying missing images..."
comm -23 "$REPORT_DIR/referenced_images.txt" "$REPORT_DIR/existing_images.txt" > "$REPORT_DIR/missing_images.txt"

# 6. Generate summary report
echo "üìä Generating summary report..."
cat << EOF > "$REPORT_DIR/audit_summary.md"
# Image Audit Summary

**Date**: $(date)
**Total Referenced Images**: $(wc -l < "$REPORT_DIR/referenced_images.txt")
**Total Existing Images**: $(wc -l < "$REPORT_DIR/existing_images.txt")
**Unused Images**: $(wc -l < "$REPORT_DIR/unused_images.txt")
**Missing Images (404s)**: $(wc -l < "$REPORT_DIR/missing_images.txt")

## Files to Review
- All references: all_image_refs.txt
- Referenced images: referenced_images.txt  
- Existing images: existing_images.txt
- Unused images: unused_images.txt
- Missing images: missing_images.txt
EOF

echo "‚úÖ Image audit complete! Reports saved to $REPORT_DIR"
